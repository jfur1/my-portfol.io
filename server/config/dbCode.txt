CREATE TABLE IF NOT EXISTS profile (
  profile_id INT NOT NULL,
  bio VARCHAR(240) NULL,
  links VARCHAR(500) NULL,
  city VARCHAR(45) NULL,
  state VARCHAR(45) NULL,
  country VARCHAR(45) NULL,
  users_user_id INT NOT NULL,
  wall_post_id INT NOT NULL,
  PRIMARY KEY (profile_id, users_user_id, wall_post_id),
  CONSTRAINT fk_profile_user
    foreign key(users_user_id)
    REFERENCES users (user_id)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION
  CONSTRAINT fk_profile_wall
    foreign key(wall_post_id)
    REFERENCES wall_posts(wall_post_id)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION
);

CREATE TABLE IF NOT EXISTS wall_posts(
  wall_post_id INT NOT NULL,
  wall_post_title VARCHAR(45) NULL,
  wall_post_content VARCHAR(240) NOT NULL,
  wall_post_date DATE NOT NULL DEFAULT CURRENT_DATE,
  users_user_id INT NOT NULL,
  profile_profile_id INT NOT NULL,
  PRIMARY KEY(wall_post_id, users_user_id, profile_profile_id),
  CONSTRAINT fk_wall_user
    foreign key(users_user_id)
    REFERENCES users(user_id)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION
  CONSTRAINT fk_profile_wall
    foreign key(profile_profile_id)
    REFERENCES profile(profile_id)
);
